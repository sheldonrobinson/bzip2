# Features that can be enabled for cmake (see CMakeLists.txt)

option(ENABLE_WERROR     "Turn on compile time warnings")

option(BZIP2_ENABLE_DEBUG      "Turn on debug output")

option(ENABLE_APP        "Build applications (bzip2, and bzip2recover)"
    ${ENABLE_APP_DEFAULT})

option(ENABLE_DOCS       "Generate documentation"
    ${ENABLE_DOCS_DEFAULT})

option(ENABLE_EXAMPLES   "Build examples"
    ${ENABLE_EXAMPLES_DEFAULT})

option(BZIP2_ENABLE_LIB_ONLY   "Build libbz2 only.  This is a short hand for -DENABLE_APP=0 -DENABLE_EXAMPLES=0")

option(BZIP2_ENABLE_STATIC_LIB "Build libbz2 in static mode also")

option(BZIP2_ENABLE_SHARED_LIB "Build libbz2 as a shared library" ON)

option(BZIP2_ENABLE_INSTALL "Install applications and library" OFF)

option(USE_OLD_SONAME "Use libbz2.so.1.0 for compatibility with old Makefiles" OFF)
